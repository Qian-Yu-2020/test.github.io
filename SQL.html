at least two cities must have multiple orders

CREATE VIEW v2 AS
SELECT t_order.orderid, orderdate, t_order.storeid, storecity
FROM t_order, t_store
WHERE t_order.storeid = t_store.storeid
ORDER BY orderid, storecity
;
SELECT * FROM v2;
